Cod C++ pentru matrice inversabilă

#include <iostream>
using namespace std;

int main() 
{
    int n;
    cout << "Introdu dimensiunea matricei n (max 10): ";
    cin >> n;
    if(n<1 || n>10)
    {
        cout << "Dimensiune nepermisa";
        return 0;
    }
    double a[10][10];
    double inv[10][10] = {0}; 
    cout << "Introdu elementele matricei:"<<endl;
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            cin >> a[i][j];
    for(int i=0;i<n;i++)
        inv[i][i] = 1;
    cout << "Matricea este:\n";
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
            cout << a[i][j] << " ";
        cout << endl;
    }
    for(int i=0;i<n;i++)
    {
        if(a[i][i] == 0)
        {
            int sr = -1;
            for(int k=i+1;k<n;k++)
            {
                if(a[k][i] != 0)
                {
                    sr = k;
                    break;
                }
            }
            if(sr == -1)
            {
                cout << "Matricea nu este inversabila."<<endl;
                return 0;
            }
            for(int j=0;j<n;j++)
            {
                swap(a[i][j], a[sr][j]);
                swap(inv[i][j], inv[sr][j]);
            }
        }
        double p= a[i][i];
        for(int j=0;j<n;j++){
            a[i][j] /= p;
            inv[i][j] /= p;
        }
        for(int k=0;k<n;k++)
        {
            if(k==i) 
                continue;
            double factor = a[k][i];
            for(int j=0;j<n;j++)
            {
                a[k][j] -= factor*a[i][j];
                inv[k][j] -= factor*inv[i][j];
            }
        }
    }
    cout << "Matricea este inversabila"<<endl;
    cout << "Inversa matricei este:"<<endl;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
            cout << inv[i][j] << " ";
        cout << endl;
    }
    return 0;
}
